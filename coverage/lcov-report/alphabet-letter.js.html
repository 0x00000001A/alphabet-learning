var Utils = require('./utils');
 
var AlphabetLetter = (function() {
  function AlphabetLetter(letter, description) {
    if (!Utils.isString(letter) || !letter.length) {
      throw new Error('Letter should be a String and should not be empty');
    }
 
    if (!Utils.isString(description) || !description.length) {
      throw new Error(
        'Letter description should be a String and should not be empty'
      );
    }
 
    this._score = 0;
    this._letter = letter;
    this._description = description;
  }
 
  AlphabetLetter.prototype = {
    constructor: AlphabetLetter,
 
    addScore: function() {
      this._score += 1;
    },
 
    getScore: function() {
      return this._score;
    },
 
    reduceScore: function() {
      this._score -= 1;
    },
 
    getLetter: function() {
      return this._letter;
    },
 
    getDescription: function() {
      return this._description;
    }
  };
 
  return AlphabetLetter;
})();
 
module.exports = AlphabetLetter;
