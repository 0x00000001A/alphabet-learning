var Utils = require('./utils');
var AlphabetLetter = require('./alphabet-letter');
var AlphabetPronounce = require('./alphabet-pronouncing');
var AlphabetLetterGroup = require('./alphabet-letter-group');
 
var Alphabet = (function() {
  var groups;
  var groupsCount;
  var languageCode;
  var alphabetPronounce;
 
  function Alphabet(_languageCode, _groups) {
    if (!Utils.isString(_languageCode)) {
      throw new Error('Language code should be a string');
    }
 
    if (!_groups) {
      groups = [];
    } else {
      if (!Utils.isArray(_groups)) {
        throw new Error('Groups is not and array');
      }
 
      if (!Utils.arrayItemsAreInstancesOf(_groups, AlphabetLetterGroup)) {
        throw new Error('Groups item should be a alphabet letter group');
      }
    }
 
    groups = _groups;
    groupsCount = groups.length;
    languageCode = _languageCode;
    alphabetPronounce = new AlphabetPronounce(languageCode);
  }
 
  Alphabet.prototype = {
    constructor: Alphabet,
 
    addGroup: function(value) {
      if (!Utils.isInstanceOf(value, AlphabetLetterGroup)) {
        throw new Error('Value should be an alphabet letter group');
      }
 
      groups.push(value);
      groupsCount = groups.length;
    },
 
    getGroup: function(id) {
      if (!Utils.isNumber(id)) {
        throw new Error('ID is incorrect');
      }
 
      return groups[id];
    },
 
    pronounce: function(alphabetLetter) {
      if (!Utils.isInstanceOf(alphabetLetter, AlphabetLetter)) {
        throw new Error('Letter should be instance of AlphabetLetter class');
      }
 
      alphabetPronounce.pronounce(alphabetLetter);
    },
 
    size: function() {
      return groupsCount;
    }
  };
 
  return Alphabet;
})();
 
module.exports = Alphabet;
