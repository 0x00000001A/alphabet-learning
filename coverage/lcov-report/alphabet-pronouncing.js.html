var Utils = require('./utils');
var AlphabetLetter = require('./alphabet-letter');
 
var AlphabetPronouncing = (function() {
  function AlphabetPronouncing(languageCode) {
    this.cache = [];
    this.apiUri =
      'https://translate.google.ru/translate_tts?ie=UTF-8&q=%s' +
      '&tl=%l&total=1&idx=0&textlen=1&client=tw-ob&ttsspeed=0.24';
    this.language = languageCode;
    this.browserSupport = false;
 
    if (!Utils.isString(languageCode)) {
      throw new Error('Language should be a string');
    }
 
    try {
      /* eslint-disable no-new */
      new Audio();
      this.browserSupport = true;
    } catch (e) {
      /* istanbul ignore next */
      this.browserSupport = false;
    }
  }
 
  AlphabetPronouncing.prototype = {
    constructor: AlphabetPronouncing,
 
    pronounce: function(letter) {
      if (!this.browserSupport) {
        // Browser does't support mp3
        return;
      }
 
      if (!Utils.isInstanceOf(letter, AlphabetLetter)) {
        throw new Error('Letter should be a string');
      }
 
      const cachedAudio = this.cache[letter.getLetter()];
 
      /* istanbul ignore next */
      if (cachedAudio) {
        cachedAudio.play();
      } else {
        this.cache[letter.getLetter()] = new Audio(
          this.apiUri
            .replace('%s', letter.getLetter())
            .replace('%l', this.language)
        );
 
        /* istanbul ignore next */
        this.cache[letter.getLetter()].oncanplay = function() {
          this.cache[letter.getLetter()].play();
        }.bind(this);
      }
    }
  };
 
  return AlphabetPronouncing;
})();
 
module.exports = AlphabetPronouncing;
