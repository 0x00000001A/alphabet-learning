var Utils = require('./utils');
var Alphabet = require('./alphabet');
 
var Quiz = (function() {
  var self;
  var view;
  var alphabet;
  var database;
  var currentQuestion;
  var currentQuestionGroup;
 
  var MIN_SCORE_FOR_PASS = 4;
  var MAX_SCORE_FOR_HELP = 2;
  var QUESTIONS_SWITCH_DELAY = 400;
 
  function initView() {
    view = {};
    view.latin = document.getElementById('quiz__latin');
    view.letter = document.getElementById('quiz__letter');
    view.options = document.getElementById('quiz__options');
    view.progress = document.getElementById('quiz__progress');
  }
 
  function updateView() {
    if (currentQuestion.getScore() >= MAX_SCORE_FOR_HELP) {
      view.latin.innerHTML = 'Select one option';
    } else {
      view.latin.innerHTML = currentQuestion.getDescription();
      alphabet.pronounce(currentQuestion);
    }
 
    view.letter.innerHTML = currentQuestion.getLetter();
    view.options.innerHTML = '';
    view.progress.style.width = getCompleteAmount() + '%';
 
    const options = currentQuestionGroup.getLetters().slice();
    shuffleOptions(options);
 
    options.forEach(char => {
      var option = document.createElement('div');
 
      Utils.addClass(option, 'quiz__option');
      option.innerHTML = char.getDescription();
 
      option.addEventListener(
        'mouseup',
        function(e) {
          this.makeChoice(option.innerHTML, e.target);
        }.bind(self)
      );
 
      view.options.appendChild(option);
    });
  }
 
  function initDatabase(_alphabet) {
    if (!Utils.isInstanceOf(_alphabet, Alphabet)) {
      throw new Error('Alphabet should be create using Alphabet class');
    }
 
    if (!_alphabet.size()) {
      throw new Error('Alphabet should not be empty');
    }
 
    alphabet = _alphabet;
    database = [];
    database.push(alphabet.getGroup(0));
  }
 
  function getCompleteAmount() {
    var a = 0;
    var total = 0;
 
    for (var u = 0; u < currentQuestionGroup.size(); u++) {
      if (currentQuestionGroup.getLetter(u).getScore() >= MIN_SCORE_FOR_PASS) {
        a += 1;
      }
 
      total +=1;
    }
 
    return a * 100 / total;
  }
 
  function isGroupComplete() {
    for (var iDb = 0; iDb < database.length; iDb++) {
      var letters = database[iDb].getLetters();
 
      for (var iCh = 0; iCh < letters.length; iCh++) {
        if (letters[iCh].getScore() < MIN_SCORE_FOR_PASS) {
          return false;
        }
      }
    }
 
    return true;
  }
 
  function rollCurrentQuestion() {
    currentQuestion = currentQuestionGroup.getLetter(
      Utils.getRandomUpTo(currentQuestionGroup.size())
    );
  }
 
  function rollCurrentQuestionGroup() {
    currentQuestionGroup = database[Utils.getRandomUpTo(database.length)];
  }
 
  function shuffleOptions(options) {
    var j, x, i;
    for (i = options.length - 1; i > 0; i--) {
      j = Math.floor(Math.random() * (i + 1));
      x = options[i];
      options[i] = options[j];
      options[j] = x;
    }
  }
 
  function next() {
    setTimeout(function() {
      if (isGroupComplete()) {
        var groupToAdd = alphabet.getGroup(database.length);
        if (groupToAdd) {
          database.push(groupToAdd);
        }
      }
 
      rollCurrentQuestionGroup();
      rollCurrentQuestion();
      updateView();
    }, QUESTIONS_SWITCH_DELAY);
  }
 
  function Quiz(_alphabet) {
    self = this;
 
    initDatabase(_alphabet);
    initView();
  }
 
  Quiz.prototype = {
    constructor: Quiz,
 
    start: function() {
      next();
    },
 
    // try using ids
    makeChoice: function(answer, elem) {
      if (currentQuestion.getDescription() === answer) {
        currentQuestion.addScore();
        Utils.addClass(elem, 'success');
        next();
      } else {
        currentQuestion.reduceScore();
        Utils.addClass(elem, 'error');
        Utils.addClass(view.letter, 'error');
 
        setTimeout(function() {
          Utils.removeClass(elem, 'error');
          Utils.removeClass(view.letter, 'error');
        }, QUESTIONS_SWITCH_DELAY);
      }
 
      alphabet.pronounce(currentQuestion);
    }
  };
 
  return Quiz;
})();
 
module.exports = Quiz;
